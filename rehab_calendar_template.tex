% Short month names used inside each day cell.
\newcommand{\MonthShortName}[1]{%
  \ifcase#1
  \or Jan%
  \or Feb%
  \or Mar%
  \or Apr%
  \or May%
  \or Jun%
  \or Jul%
  \or Aug%
  \or Sep%
  \or Oct%
  \or Nov%
  \or Dec%
  \else ?%
  \fi
}

% Finds the number of days in the current month, including leap years.
\newcommand{\ComputeCurrentMonthDays}{%
  \CurrentMonthDays=31
  \ifnum\CurrentMonth=4 \CurrentMonthDays=30\fi
  \ifnum\CurrentMonth=6 \CurrentMonthDays=30\fi
  \ifnum\CurrentMonth=9 \CurrentMonthDays=30\fi
  \ifnum\CurrentMonth=11 \CurrentMonthDays=30\fi
  \ifnum\CurrentMonth=2
    \CurrentMonthDays=28
    \TempCount=\numexpr\the\CurrentYear-400*(\the\CurrentYear/400)\relax
    \ifnum\TempCount=0
      \CurrentMonthDays=29
    \else
      \TempCount=\numexpr\the\CurrentYear-100*(\the\CurrentYear/100)\relax
      \ifnum\TempCount=0
      \else
        \TempCount=\numexpr\the\CurrentYear-4*(\the\CurrentYear/4)\relax
        \ifnum\TempCount=0
          \CurrentMonthDays=29
        \fi
      \fi
    \fi
  \fi
}

\newcommand{\ComputeCurrentWeekday}{%
  % Zeller's congruence converted to Monday=1 through Sunday=7.
  \AdjustedYear=\CurrentYear
  \AdjustedMonth=\CurrentMonth
  \ifnum\AdjustedMonth<3
    \advance\AdjustedMonth by 12
    \advance\AdjustedYear by -1
  \fi
  \Century=\numexpr\the\AdjustedYear/100\relax
  \YearOfCentury=\numexpr\the\AdjustedYear-100*\the\Century\relax
  \TempCount=\numexpr
    \the\CurrentDay + (13*(\the\AdjustedMonth+1))/5 + \the\YearOfCentury
    + \the\YearOfCentury/4 + \the\Century/4 + 5*\the\Century
  \relax
  \CurrentWeekday=\numexpr\the\TempCount-7*(\the\TempCount/7)\relax
  \advance\CurrentWeekday by -1
  \ifnum\CurrentWeekday<1
    \CurrentWeekday=7
  \fi
}

% Advances both the date and the weekday to the next calendar day.
\newcommand{\AdvanceCurrentDate}{%
  \ComputeCurrentMonthDays
  \ifnum\CurrentWeekday<7
    \global\advance\CurrentWeekday by 1
  \else
    \global\CurrentWeekday=1
  \fi
  \global\advance\CurrentDay by 1
  \ifnum\CurrentDay>\CurrentMonthDays
    \global\CurrentDay=1
    \global\advance\CurrentMonth by 1
    \ifnum\CurrentMonth>12
      \global\CurrentMonth=1
      \global\advance\CurrentYear by 1
    \fi
  \fi
}

% Resolves checkbox slot 1-4 to its display label.
\newcommand{\CheckboxLabelFor}[1]{%
  \ifcase#1
  \or \CheckboxLabelOne%
  \or \CheckboxLabelTwo%
  \or \CheckboxLabelThree%
  \or \CheckboxLabelFour%
  \else ?%
  \fi
}

% Pulls out a single character from a 7-character weekday pattern.
\def\PatternCharAt#1#2{%
  \expandafter\PatternCharAtAux#1\relax{#2}%
}

\def\PatternCharAtAux#1#2#3#4#5#6#7\relax#8{%
  \ifcase#8 ?%
  \or #1%
  \or #2%
  \or #3%
  \or #4%
  \or #5%
  \or #6%
  \or #7%
  \else ?%
  \fi
}

% Returns 1 when the pattern enables the checkbox on that weekday.
\newcommand{\PatternEnabledFor}[2]{%
  \if\expandafter\PatternCharAt\expandafter{#1}{#2}-0\else1\fi
}

% Chooses the correct pattern for checkbox slot 1-4.
\newcommand{\CheckboxEnabledFor}[2]{%
  \ifcase#1 0%
  \or \PatternEnabledFor{\CheckboxPatternOne}{#2}%
  \or \PatternEnabledFor{\CheckboxPatternTwo}{#2}%
  \or \PatternEnabledFor{\CheckboxPatternThree}{#2}%
  \or \PatternEnabledFor{\CheckboxPatternFour}{#2}%
  \else 0%
  \fi
}

% Checks whether the current weekday should show any checkbox at all.
\newcommand{\AnyCheckboxEnabledFor}[1]{%
  \ifnum\CheckboxEnabledFor{1}{#1}>0
    1%
  \else
    \ifnum\CheckboxEnabledFor{2}{#1}>0
      1%
    \else
      \ifnum\CheckboxEnabledFor{3}{#1}>0
        1%
      \else
        \ifnum\CheckboxEnabledFor{4}{#1}>0
          1%
        \else
          0%
        \fi
      \fi
    \fi
  \fi
}

% Renders the label row and square row in aligned columns.
\newcommand{\CheckboxLabelRow}[1]{%
  \makebox[\linewidth][l]{%
    \ifnum\CheckboxEnabledFor{1}{#1}>0
      \makebox[\CheckboxColumnWidth][c]{\scriptsize\textbf{\CheckboxLabelFor{1}}}%
    \fi
    \ifnum\CheckboxEnabledFor{2}{#1}>0
      \makebox[\CheckboxColumnWidth][c]{\scriptsize\textbf{\CheckboxLabelFor{2}}}%
    \fi
    \ifnum\CheckboxEnabledFor{3}{#1}>0
      \makebox[\CheckboxColumnWidth][c]{\scriptsize\textbf{\CheckboxLabelFor{3}}}%
    \fi
    \ifnum\CheckboxEnabledFor{4}{#1}>0
      \makebox[\CheckboxColumnWidth][c]{\scriptsize\textbf{\CheckboxLabelFor{4}}}%
    \fi
  }\par
}

\newcommand{\CheckboxSquareRow}[1]{%
  \makebox[\linewidth][l]{%
    \ifnum\CheckboxEnabledFor{1}{#1}>0
      \makebox[\CheckboxColumnWidth][c]{\scalebox{1.25}{$\square$}}%
    \fi
    \ifnum\CheckboxEnabledFor{2}{#1}>0
      \makebox[\CheckboxColumnWidth][c]{\scalebox{1.25}{$\square$}}%
    \fi
    \ifnum\CheckboxEnabledFor{3}{#1}>0
      \makebox[\CheckboxColumnWidth][c]{\scalebox{1.25}{$\square$}}%
    \fi
    \ifnum\CheckboxEnabledFor{4}{#1}>0
      \makebox[\CheckboxColumnWidth][c]{\scalebox{1.25}{$\square$}}%
    \fi
  }%
}

% Builds the checkbox block for the current day cell.
\newcommand{\DayCheckboxes}{%
  \ifnum\AnyCheckboxEnabledFor{\the\CurrentWeekday}>0
    \CheckboxLabelRow{\the\CurrentWeekday}%
    \vspace{2pt}%
    \CheckboxSquareRow{\the\CurrentWeekday}%
  \fi
}

% Blank filler cells keep the Monday-Sunday table aligned at the start/end.
\newcommand{\BlankCellContent}{%
  \parbox[t][\DayCellHeight][t]{\linewidth}{\mbox{}}%
}

% Active day cells show the date plus any enabled checkbox labels and squares.
\newcommand{\ActiveCellContent}{%
  \parbox[t][\DayCellHeight][t]{\linewidth}{%
    \raggedright
    \vspace{-4pt} % how far down the whole block starts
    {\Large\bfseries \the\CurrentDay}\par
    {\small\textit{\MonthShortName{\the\CurrentMonth}}}\par
    \vspace{6pt} % vertical gap between date and check boxes
    \DayCheckboxes
  }%
}

% Emits one table cell, choosing between leading blank, active day, or trailing blank.
\newcommand{\NextCalendarCell}{%
  \ifnum\PendingLeadingBlanks>0
    \global\advance\PendingLeadingBlanks by -1
    \BlankCellContent
  \else
    \ifnum\DaysRemaining>0
      \ActiveCellContent
      \global\advance\DaysRemaining by -1
      \AdvanceCurrentDate
    \else
      \BlankCellContent
    \fi
  \fi
}

% Writes complete 7-column rows until the calendar is fully rendered.
\newcommand{\EmitCalendarRows}[1]{%
  \ifnum#1<1
  \else
    \NextCalendarCell &
    \NextCalendarCell &
    \NextCalendarCell &
    \NextCalendarCell &
    \NextCalendarCell &
    \NextCalendarCell &
    \NextCalendarCell \\
    \hline
    \expandafter\EmitCalendarRows\expandafter{\number\numexpr#1-1\relax}%
  \fi
}

  }%
